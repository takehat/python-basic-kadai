array = ["水", "金", "地", "火", "木", "土", "天", "海", "冥"]

# for文を使って、配列arrayの要素を1つずつ順番に出力する
for planet in array:
  print(planet)

# while文を使って、配列arrayの要素を1つずつ順番に出力する

# 変数targetを用意する
target = "冥"

# カウンタ変数iを用意する
i = 0

# 変数planetに初期値を与える
planet = array[i]

# 変数planet2が"冥"以外である間、変数planet2の値を出力し続ける
while planet != target:
  # 変数planetを更新
  planet = array[i]
  # 現在の変数planetを表示
  print(planet)
  # カウンタ変数の値を1増やす
  i = i + 1
